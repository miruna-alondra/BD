CREATE TABLE PRODUSE
(
ID_PRODUS NUMBER(6) PRIMARY KEY,
DENUMIRE VARCHAR2(20),
DESCRIERE VARCHAR2(20),
PRET NUMBER(7,2),
STOC NUMBER(4)
);
CREATE TABLE USERI (
ID_USER NUMBER(5) PRIMARY KEY,
NUME VARCHAR2(20),
PRENUME VARCHAR2(20),
LOCALITATE VARCHAR2(20)
);

CREATE TABLE COMENZI (
ID_COMANDA NUMBER(6) PRIMARY KEY,
ID_PROD NUMBER(6),
CANTITATE NUMBER(4),
PRET_TOTAL NUMBER(9,2),
ID_USER NUMBER(5),
CONSTRAINT FK_COMENZI_PRODUSE FOREIGN KEY(ID_PROD)
REFERENCES PRODUSE(ID_PRODUS),
CONSTRAINT FK_COMENZI_USERI FOREIGN KEY(ID_USER)
REFERENCES USERI(ID_USER)
);
